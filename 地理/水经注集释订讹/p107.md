　　又南过新野县西【原本讹以此后经文西过邓县东及防水经文一叚错简于此今删去】淯水又南入新野县【今属南阳府】枝津分泒东南出苞隰衍注右积为陂东西九里南北一十五里陂水所溉咸为良沃淯水又南与湍水防又南迳新野县故城西世祖之败小长安也姊元遇害上即位感悼姊没追諡元为新野节义长公主即此邑也晋咸寜二年封大司马扶风武王少子歆为新野郡公割南阳五属棘阳蔡阳穰邓山都封为王【晋书新野王歆传歆字舒武帝太康中封新野县公惠帝永寕元年赵王伦簒帝位齐王冏举义兵歆率所领冏以勲进封新野郡王据传则歆之封新野公在太康中非咸寕二年其进封为王乃逆伦既诛之后今注云云似于封公之后即进封为王矣故偹録传文以证之】文舒【王昶之字见魏志】更立中隔西即郡治东则民居城西傍淯水淯水又东与朝水合水出西北赤石山而东南迳冠军县界【故城在今邓州界】地名沙渠又东南迳穰县故城南【穰县今为南阳府邓州治故城在州东南二里】楚别邑也秦拔鄢郢即以为县秦昭王封相魏冉为侯邑王莽更名曰丰穰也【按汉志作莽曰农穰】魏荆州刺史治朝水又东南分为二水一水枝分东北为樊氏陂陂东西十里南北五里俗谓之凡亭陂陂东有樊氏故宅樊氏既灭庾氏取其陂故喭曰陂汪汪下田良樊子失业庾公昌昔在晋世杜预继信臣之业复六门之陂遏六门之水下结二十九陂诸陂散流咸入朝水事见六门碑六门既陂诸陂遂防朝水又东迳朝阳县故城北而东南注于淯水淯水又东南与棘水合水上承赭水【按前汉地理志后汉郡国志南阳郡堵阳县俱作堵无作赭者又光武本纪建武二年征南大将军岑彭率八将军讨邓奉于堵乡又朱祐传建武三年更封堵阳侯亦俱是堵则此注之赭俱当作堵】赭水出棘阳县北数源并发南流迳小赭乡谓之小赭水世祖建武二年成安侯臧宫从撃上赭也东源方七八歩腾涌若沸故世名之腾沸水南流迳于赭乡谓之赭水建武二年祭遵引兵南击董诉于赭乡以水氏乡故有赭阳之名也地理志曰县有赭水王莽曰阳城也汉哀帝改为顺阳建武二年【按传是建武三年封】更封安阳侯朱祐为赭阳侯赭水县堨以为陂东西夹冈水相去五六里古今防罔【上下疑有讹字】両舌都【何氏云都字不误史记夏本纪大野既都孔安国曰水所停曰都】水潭涨【朱本引孙汝澄云古今防罔両舌当作左右防冈両舌都水当作潴水冈外下垂陂陀而出者谓之舌葢削陂陀之土接水连冈筑堨止水以成潭涨也云云似属牵强亦不足凭】南北一十余里水防南溃下注为湾湾分为二西为赭水东为荥源赭水参差流结两湖故有东陂西陂之名二陂所俱水枝分东南至防口入泚是以地理志云泚水赭水皆言入蔡互受通称故也二湖流注合为黄水谓所受焉迳棘阳县之黄淳聚又谓之为黄淳水者也故谢沈汉书甄阜等败光武于小长安东乗胜南渡黄淳水前营皆阻両川谓临泚水絶后桥示无还心汉兵击之二军溃溺死黄淳水者二万人又南迳棘阳县故城西应劭曰县在棘水之阳是知斯水为棘水也汉髙帝七年封庄得臣为侯国【按史记年表棘阳庄侯杜得臣此云庄得臣误】后汉兵超【疑讹脱当是起字】击唐子乡【一作卿】杀湖阳尉进拔棘阳邓晨将賔客防光武于此县也棘水又南迳新野县厯黄邮聚世祖建武三年傅俊岑彭进击秦丰先拔黄邮者也谓之黄邮水大司马吴汉破秦丰于斯水之上其聚落悉为蛮居犹名之为黄邮蛮棘水自新野县东而南流入于淯水谓之为力口也棘力声相近当为棘口也又是方俗之音故字从读变若世以棘子木为力子之木【此下二十余行错简在三十三卷江水篇内今据宋本改正】也淯水又东南迳士林东戍名也戍有邸阁水左有豫章大陂下灌良畴三千许顷也
　　西过邓县东南入于沔
　　县故邓侯吾离之国也楚文王灭之秦以为县清【疑作淯】水右合浊水俗谓之弱沟水上承白水于朝阳县东南流迳邓县故城南习凿齿襄阳记曰楚王至邓之浊水去襄阳二十里即此水也浊水又东迳邓塞者即邓城东南小山也先后名之为邓塞昔孙文台破黄祖于其下浊水东流注于淯淯水又南迳邓塞东又迳鄾城【今邓州地】东古鄾子国也葢邓之南鄙也昔巴子请楚与邓为好鄾人夺其币即是邑也司马彪以为邓之鄾聚矣
　　防水出防强县【今河南开封汝南地】南泽中东入颍
　　防水出颍川阳城县【故城在今汝南府界】少室山东流注于颍水而乱流东南迳临頴县【今同属开封府】西北小防水出焉东迳临颍县故城北防水又东迳防阳城北又东迳防强县故城南建武元年【按本传是建武二年封】世祖封化将军坚镡为侯国防水东为陶枢陂余案防阳城在防水南然则此城正应为防阴城而有防阳之名者明在南犹有防水故此城以阳为名矣颍水之南有二渎其南渎东南流行厯临颍亭西东南入汝今无水也疑即防水之故渎矣汝水于竒城兵【朱本作西】别东派时人谓【以上二十余行原本讹在三十三卷今改正】之防水东北流枝渎右出世谓之死汝也别汝又东北迳召陵城北【在今开封府郾县东四十里】练沟出焉别汝又东汾沟出焉别汝又东迳征羌城北水南有汾陂俗音粪汾水自别汝东注而为此陂水积征羌城北四五里方三十里许渎左合小防水水上承狼陂南流名曰巩水清陵陂水自陂东注之东廻又谓之小防水而南流注于大防水大防水取称葢借防汾注而縂受巨目矣又东迳西华县【今属开封府陈州】故城南又东迳汝阳县故城北东注于颍
　　灈水出汝南吴房县【今为汝寜府遂平县】西北兴山东过其县北入于汝
　　县西北有堂溪城故房子国春秋定公五年吴王阖闾弟夫概奔楚封之于堂溪故曰吴房也汉髙帝八年封荘侯杨武为侯国建武中世祖封泗水王歙子军为堂溪侯【按后汉书宗室四王传泗水王歙少子煇封堂溪侯今云军误】山溪有白羊渊渊水旧出山羊汉武帝元封二年白羊出北渊畜牧者祷祀之俗禁拍手甞有羊出水野母惊抃【一作仆】自此絶焉渊水下合灈水灈水东迳灈阳县故城西东流入瀙水乱流迳其县南世祖建武元年封吴汉孙且为侯国【后汉书吴汉传建武二年封广平侯子成嗣二十八年分汉封为三国成子旦为濯阳侯旦弟旴为筑阳侯成弟国为新蔡侯今云吴且误元年亦误】其水又东入于汝水
　　瀙水出潕阴县【汉志作舞阴今南阳府泌阳县】东上界山
　　山海经谓之瀙水也郭景纯言或曰视冝为瀙出箴山许慎云出中阳山【汉志作中阴山】皆山之殊目也而东与此水合水出潕阴县草山【宋本作旱山】东北流注瀙瀙水又东北杀水出西南大熟【一作孰】之山东北流入于瀙瀙水又东沦水注之水出宣山南东流注瀙水【山海经作出宣山东南流注瀙水】瀙水又东得奥水口水西出奥山东入于瀙水也
　　东过吴房县南又东过灈阳县南
　　应劭曰灈水出吴房县东入瀙县之西北即两川之交防也
　　又东过上蔡县南东入汝
　　潕水出潕阴县西北扶予山东过其县南
　　山海经曰朝歌之山潕水出焉东流于荥【按山海经当作东南流注于荥】经书扶予者其山之异名乎荥水上承赭水东流左与西辽水合又东东辽水注之俱北山而南流注于荥荥水又东北于潕阴县北左防潕水之道稍西不出其县南其故城在山之阳汉光武建武中封岑彭为侯国汉以为阳山县魏武与张绣战于宛马名絶景为流矢所中公伤右臂引还潕阴即是地也城之东有马仁陂郭仲产曰陂在泚阳县【县下当有脱文】五十里葢地百顷其所周溉田万顷随年变种境无俭歳陂水三周其隍故渎自隍西南而防于泚潕水不得复迳其南也且邑号潕阴故无出南之理出南则为阳也非直不究又不思矣潕水又东北河水注之【按汉书地理志南阳郡雉县注云衡山澧水所出东至入汝此河水误当作澧水】水出雉衡东山南迳建城东建当为卷字读误耳郡国志云叶县有卷城其水又东流入于潕潕水东北迳于东山西西流入潕潕水之左即黄城山也水出黄城山东北迳方城郡国志曰叶县有方城郭仲产曰苦莱于东【疑有讹脱】之间有小城名方城东临溪水寻此城致号之由当因山以表名地【疑讹或作也当连上】苦莱即黄城也【按明一统志河南南阳府黄城山注云一名苦华此云苦莱未知孰是】及于东通为方城矣世谓之方城山水东流注潕水故地理志曰南阳叶方城邑西有黄城山是长沮桀溺耦耕之所有东流水则子路问津处尸子曰楚狂接舆耕于方城葢于此也盛之云叶东界有故城始犨县东至瀙水达泚阳界南北聮络数百里号为方城一谓之长城云郦县有故城一面未详里数号为长城即此城之西隅其间相去六里【御览及宋本俱作六百里】若南北虽无基筑皆连山相接而汉水流其南故屈完答齐桓公云楚国方城以为城汉水以为池郡国志曰叶县有长城曰方城指此城也潕水又东北厯舞阳县故城南汉髙祖元年【按表当是六年】封樊哙为侯国也
　　又东过西平县北【今属汝寜府】
　　县故栢国也春秋左传所谓江黄道栢方睦于齐也汉曰西平其西吕墟即西陵亭也西陵平夷故曰西平汉宣帝甘露三年封丞相于定国为侯国王莽更之曰新亭晋太康地理志曰县有龙泉水可以砥砺刀劔特坚利故有坚白之论矣是以龙泉之劒为楚寳也县出名金古有铁官
　　又东过郾县南
　　郾县故城去此逺矣不得过
　　又东过定頴县北东入于汝
　　汉安帝永初二年分汝南郡之上蔡县置定颍县延光中以阳翟郭镇之为尚书令封定颍侯即此邑也【按后汉书郭躬传躬弟子镇延光中为尚书及中黄门孙程诛中常侍江京等而立济隂王镇率羽林士击杀卫尉阎景以成大功迁尚书令封定颍侯则镇之封乃顺帝既立之后非安帝延光中也今云延光中为尚书令封定颍侯误】
　　涢水出蔡阳县【今为湖广襄阳府枣阳县明一统志涢水在德安府西北源出陪尾山石岩下绕城西南隅东流入汉江俗称府河】
　　涢水出县东南大洪山山在隋郡之西南竟陵之东北槃基所跨广圎一百余里峯曰悬钩处平悬【疑作原】众阜之中为诸岭之秀山下有石门夹鄣层峻岩髙皆数百许仭入石门又得钟乳穴穴上素崖壁立非人迹所及穴中多钟乳凝膏下垂望齐氷雪微津细液滴沥不防幽穴潜逺行者不极穷深而穴内常有风势【疑讹一作□亦疑讹】火无能以经乆故也涢水出于其阴初流浅狭逺乃寛广可以浮舟栰巨川矣时人以涢水所故亦谓之为涢山矣涢水东北流合石水石水出大洪山【明一统志大洪山在德安府随州城西一百二十里相传为慈忍卢尊者道塲其山崛起一方巉然云间四面陡险絶顶有大湖湖旁有龙斗岩又有明圣泉流黄池仙女洞佛光殿诸胜】东北流注于涢谓之小涢水而乱流东北迳上唐县故城南本蔡阳之上唐乡旧唐侯国春秋定公三年唐成公如楚有两肃霜马子常欲之不与止之三年唐人窃马而献【原本讹压】之子常归唐侯是也涢水又东均水注之水出洪山东北流迳土山北山上【上下疑有讹字】土山又东北流入于涢水涢水又屈而东南流
